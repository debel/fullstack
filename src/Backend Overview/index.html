running a http server in node.js


          /* server.js */
          var http = require('http'),
              server = http.createServer(function (req, res) {
                  //server executes callback on every connection
                  res.end('Hello Web!');
          });

          //start the server on port 8080
          server.listen(8080);
run the server from the console

          $ node server.js
